% Content-Specific Sections
\section{Definitions}
\begin{itemize}
    \item \textbf{Environmental Hazard (OHS-Related):} An environmental condition (e.g., chemical spill, poor air quality) that poses a risk to worker health and safety.
    \item \textbf{Waste Management:} The handling, storage, and disposal of waste to minimize OHS risks.
    \item \textbf{Spill Response:} Procedures to address accidental releases of hazardous substances impacting OHS.
\end{itemize}

\section{Policy Statement}
\clientName\ is committed to:
\begin{itemize}
    \item \textbf{Environmental Risk Management:} Identifying and controlling environmental hazards that impact OHS, such as chemical spills, air pollution, or improper waste handling.
    \item \textbf{Legal Compliance:} Complying with South African legislation (e.g., OHS Act Section 8, Environmental Regulations for Workplaces) and international standards (e.g., SANS 45001 Clause 8.1.2).
    \item \textbf{Hazardous Substance Handling:} Ensuring safe storage, use, and disposal of hazardous substances to protect worker health, per Hazardous Chemical Substances Regulations.
    \item \textbf{Air Quality Management:} Monitoring and maintaining workplace air quality to prevent health risks (e.g., respiratory issues).
    \item \textbf{Waste Management:} Implementing safe waste handling practices to minimize OHS risks (e.g., exposure to hazardous waste).
    \item \textbf{Spill Response:} Establishing procedures to respond to environmental incidents that affect OHS, ensuring worker safety.
    \item \textbf{Training and Awareness:} Providing training on OHS-related environmental risks and safe practices.
    \item \textbf{Continual Improvement:} Regularly reviewing and improving environmental management practices to enhance OHS outcomes.
    \item \textbf{Stakeholder Engagement:} Consulting with employees and contractors on environmental OHS issues.
\end{itemize}

\begin{itemize}
    \item \textbf{Top Management (CEO, {{CEO_NAME}}):}

    \item \textbf{OHS Manager (\compilerName):}

    \end{itemize}
    \item \textbf{Line Managers:}
    \begin{itemize}
        \item Implementing environmental controls in their areas.
        \item Ensuring compliance with waste and spill response procedures.
    \end{itemize}
    \item \textbf{Employees:}
    \begin{itemize}
        \item Adhering to environmental safety procedures.
        \item Reporting environmental hazards impacting OHS.
    \end{itemize}

\section{Key Principles}
\begin{itemize}
    \item \textbf{Risk-Based Approach:} Prioritizing controls based on OHS-related environmental risks.
    \item \textbf{Prevention:} Proactively managing environmental hazards to protect health and safety.
    \item \textbf{Responsibility:} Ensuring all employees contribute to safe environmental practices.
\end{itemize}

\section{Legal and Standard Framework}
This policy is guided by:
\begin{itemize}
    \item Occupational Health and Safety Act 85 of 1993 (Section 8).
    \item Environmental Regulations for Workplaces, 1987.
    \item Hazardous Chemical Substances Regulations (HCSR).
    \item SANS 45001: Occupational Health and Safety Management Systems (Clause 8.1.2).
\end{itemize}

\section{Policy Review}
This policy will be reviewed at least every two years or as necessitated by changes in legislation, operations, or environmental risks.
