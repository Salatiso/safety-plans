% Content-Specific Sections
\section{Definitions}
\begin{itemize}
    \item \textbf{Contractor:} An external individual or company engaged to perform work for \clientName.
    \item \textbf{OHS Compliance:} Adherence to OHS legislation, standards, and \clientName\ policies.
\end{itemize}

\section{Policy Statement}
\clientName\ is committed to:
\begin{itemize}
    \item \textbf{Contractor Selection:} Selecting contractors based on their OHS performance and compliance records.
    \item \textbf{Induction:} Providing OHS induction to contractors before work commences.
    \item \textbf{Monitoring:} Regularly monitoring contractors’ OHS performance through inspections and audits.
    \item \textbf{Legal Compliance:} Ensuring contractors comply with OHSA 85 of 1993 and Construction Regulations 2014.
    \item \textbf{Performance Evaluation:} Assessing contractors’ OHS performance and addressing non-compliances.
    \item \textbf{Communication:} Maintaining open communication with contractors on OHS matters.
\end{itemize}

\section{Roles and Responsibilities}
\begin{itemize}
    \item \textbf{Top Management (CEO, {{CEO_NAME}}):}
    \begin{itemize}
        \item Ensuring resources for contractor management processes.
        \item Approving contractor selection criteria.
    \end{itemize}
    \item \textbf{OHS Manager (\compilerName):}
    \begin{itemize}
        \item Developing contractor management procedures.
        \item Coordinating inductions and monitoring compliance.
    \end{itemize}
    \item \textbf{Project Managers:}
    \begin{itemize}
        \item Overseeing contractors’ daily OHS performance.
        \item Reporting non-compliances to the OHS Manager.
    \end{itemize}
    \item \textbf{Contractors:}
    \begin{itemize}
        \item Complying with \clientName\ OHS policies and legal requirements.
        \item Participating in inductions and inspections.
    \end{itemize}
\end{itemize}

\section{Key Principles}
\begin{itemize}
    \item \textbf{Proactive Management:} Address OHS risks before contractors begin work.
    \item \textbf{Accountability:} Hold contractors accountable for OHS performance.
    \item \textbf{Collaboration:} Work with contractors to improve OHS outcomes.
\end{itemize}

\section{Legal and Standard Framework}
This policy is guided by:
\begin{itemize}
    \item Occupational Health and Safety Act 85 of 1993 (Section 37).
    \item Construction Regulations, 2014.
    \item SANS 45001: Occupational Health and Safety Management Systems (Clause 8.1.4).
\end{itemize]

\section{Policy Review}
This policy will be reviewed at least every two years or as necessitated by changes in legislation or contractor performance.
