% Content-Specific Sections
\section{Definitions}
\begin{itemize}
    \item \textbf{Audit:} A systematic, independent examination to determine whether OHS activities comply with planned arrangements.
    \item \textbf{Inspection:} A routine check of workplace conditions to identify hazards and non-compliances.
    \item \textbf{Non-Conformance:} Failure to meet OHS requirements or standards.
\end{itemize}

\section{Policy Statement}
\clientName\ is committed to:
\begin{itemize}
    \item \textbf{Conducting Audits:} Performing regular internal and external audits to verify OHS compliance.
    \item \textbf{Performing Inspections:} Conducting routine workplace inspections to identify hazards.
    \item \textbf{Legal Compliance:} Ensuring audits and inspections align with OHSA 85 of 1993 and SANS 45001.
    \item \textbf{Corrective Actions:} Addressing non-conformances promptly through corrective action plans.
    \item \textbf{Training:} Providing training for auditors and inspectors.
    \item \textbf{Continual Improvement:} Using audit and inspection findings to enhance OHS performance.
\end{itemize}

\section{Roles and Responsibilities}
\begin{itemize}
    \item \textbf{Top Management (CEO, {{CEO_NAME}}):}
    \begin{itemize}
        \item Ensuring resources for audits and inspections.
        \item Reviewing audit findings and approving corrective actions.
    \end{itemize}
    \item \textbf{OHS Manager (\compilerName):}
    \begin{itemize}
        \item Developing audit and inspection schedules.
        \item Coordinating audits and ensuring follow-up actions.
    \end{itemize}
    \item \textbf{Auditors and Inspectors:}
    \begin{itemize}
        \item Conducting audits and inspections per defined procedures.
        \item Reporting findings and recommending improvements.
    \end{itemize}
    \item \textbf{Employees:}
    \begin{itemize}
        \item Participating in audits and inspections.
        \item Reporting hazards and non-compliances.
    \end{itemize}
\end{itemize}

\section{Key Principles}
\begin{itemize}
    \item \textbf{Systematic Approach:} Use structured processes for audits and inspections.
    \item \textbf{Independence:} Ensure auditors are independent of the areas audited.
    \item \textbf{Proactivity:} Identify issues before they cause harm.
\end{itemize}

\section{Legal and Standard Framework}
This policy is guided by:
\begin{itemize}
    \item Occupational Health and Safety Act 85 of 1993.
    \item SANS 45001: Occupational Health and Safety Management Systems (Clause 9.2).
    \item ISO 19011: Guidelines for Auditing Management Systems.
\end{itemize}

\section{Policy Review}
This policy will be reviewed at least every two years or as necessitated by changes in legislation or audit findings.
