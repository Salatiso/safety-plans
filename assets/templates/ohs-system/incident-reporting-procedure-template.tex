% Content-Specific Sections
\section{Definitions}
\begin{itemize}
    \item \textbf{Incident:} Any event that results in, or could have resulted in, injury, illness, damage to property, or environmental harm.
    \item \textbf{Near Miss:} An incident that could have caused harm but did not.
    \item \textbf{COIDA:} Compensation for Occupational Injuries and Diseases Act 130 of 1993.
\end{itemize}

\section{Procedure}

\subsection{Immediate Response}
\begin{itemize}
    \item Ensure the safety of all personnel by removing them from danger.
    \item Provide first aid or medical attention as needed.
    \item Secure the incident scene to preserve evidence.
\end{itemize}

\subsection{Incident Reporting}
\begin{itemize}
    \item \textbf{Who Reports:} All employees, contractors, and visitors must report incidents immediately.
    \item \textbf{To Whom:} Notify the immediate supervisor or OHS Manager (\compilerName, \compilerPhone).
    \item \textbf{How:} Complete the Incident Report Form (see template in SafetyHelp forms library) within 24 hours.
\end{itemize}

\subsection{Notification}
\begin{itemize}
    \item \textbf{Internal Notification:} The OHS Manager informs senior management and the OHS Committee.
    \item \textbf{External Notification:} Report to the Department of Labour within 7 days for reportable incidents (COIDA Section 24, OHSA Section 24).
\end{itemize}

\subsection{Investigation}
\begin{itemize}
    \item Assign an investigation team, including an OHS representative.
    \item Identify root causes and contributing factors.
    \item Document findings in an Incident Investigation Report.
\end{itemize}

\subsection{Corrective Actions}
\begin{itemize}
    \item Implement corrective actions to prevent recurrence.
    \item Monitor and verify the effectiveness of actions.
\end{itemize}

\section{Legal Compliance}
This procedure ensures compliance with:
\begin{itemize}
    \item Occupational Health and Safety Act 85 of 1993 (Section 24).
    \item Compensation for Occupational Injuries and Diseases Act 130 of 1993 (Section 24).
    \item SANS 45001 Clause 10.2: Incident, Nonconformity, and Corrective Action.
\end{itemize}
